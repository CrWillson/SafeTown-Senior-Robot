18#ifndef RotaryEncoder_h
19#define RotaryEncoder_h
20
21#include "Arduino.h"
22
23class RotaryEncoder
24{
25public:
26 enum class Direction {
27 NOROTATION = 0,
28 CLOCKWISE = 1,
29 COUNTERCLOCKWISE = -1
30 };
31
32 enum class LatchMode {
33 FOUR3 = 1, // 4 steps, Latch at position 3 only (compatible to older versions)
34 FOUR0 = 2, // 4 steps, Latch at position 0 (reverse wirings)
35 TWO03 = 3 // 2 steps, Latch at position 0 and 3
36 };
37
38 // ----- Constructor -----
39 RotaryEncoder(int pin1, int pin2, LatchMode mode = LatchMode::FOUR0);
40
41 // retrieve the current position
42 long getPosition();
43
44 // simple retrieve of the direction the knob was rotated last time. 0 = No rotation, 1 = Clockwise, -1 = Counter Clockwise
45 Direction getDirection();
46
47 // adjust the current position
48 void setPosition(long newPosition);
49
50 // call this function every some milliseconds or by using an interrupt for handling state changes of the rotary encoder.
51 void tick(void);
52
53 // Returns the time in milliseconds between the current observed
54 unsigned long getMillisBetweenRotations() const;
55
56 // Returns the RPM
57 unsigned long getRPM();
58
59private:
60 int _pin1, _pin2; // Arduino pins used for the encoder.
61
62 LatchMode _mode; // Latch mode from initialization
63
64 volatile int8_t _oldState;
65
66 volatile long _position; // Internal position (4 times _positionExt)
67 volatile long _positionExt; // External position
68 volatile long _positionExtPrev; // External position (used only for direction checking)
69
70 unsigned long _positionExtTime; // The time the last position change was detected.
71 unsigned long _positionExtTimePrev; // The time the previous position change was detected.
72};
73
74#endif
75
